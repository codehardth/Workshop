// cookieManager.ts (refactored)
export function setSessionCookie(
  res: Response,
  token: string
): void {
  const isProduction = process.env.NODE_ENV === 'production';

  res.setHeader('Set-Cookie', [
    `session=${token}`,
    'HttpOnly',
    'Path=/',
    'SameSite=Strict',
    isProduction ? 'Secure' : '',
  ].filter(Boolean).join('; '));
}

// ‚úÖ All tests pass + Production ready!
